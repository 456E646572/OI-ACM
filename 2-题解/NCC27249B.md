## [Winding Number](https://ac.nowcoder.com/acm/contest/27249/B)

给定一个多边形，求 $P$ 点相对于该多边形的回转数。
先特判 $P$ 点在多边形边上的情况。然后使用光线回转法（Sunday's algorithm），从 $P$ 点出发向 $x$ 轴正方向引一条射线，当多边形的边从下往上穿过射线时，回转数加一，从上往下穿过射线时，回转数减一。

#### [AC 代码]()

```cpp
#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
const int inf=INT_MAX;
int sgn(ll x) { return x==0? 0: ( x>0? 1: -1); }

struct Point {
    ll x, y;
    Point(ll x=0, ll y=0): x(x), y(y) { }
    void in() { cin >> x >> y; }
    Point operator + (const Point &p) const {
        return Point(x+p.x, y+p.y);
    }
    Point operator - (const Point &p) const {
        return Point(x-p.x, y-p.y);
    }
    ll operator * (const Point &p) const {
        return x*p.x + y*p.y;
    }
    ll operator ^ (const Point &p) const {
        return x*p.y - y*p.x;
    }
};
typedef Point Vector;

struct Line {
    Point a, b;
    Vector v;
    Line(Point a, Point b): a(a), b(b) { 
        v = b - a;
    }
    int toLeft(Point p) {
        return sgn(v^(p-a));
    }
    bool onSegment(Point p) {
        return (toLeft(p)==0) && ((p-a)*(p-b)<=0);
    }
};

struct Polygon {
    int n;
    vector <Point> poly;
    Polygon(int n): n(n) {
        poly.resize(n);
        for (auto &p: poly) { p.in(); }
    }
    int inPolygon(Point p) {
        int wn = 0;
        for (int i = 0; i < n; ++i) {
            if (Line(poly[i], poly[(i + 1) % n]).onSegment(p)) { return -inf; }
            int k = Line(poly[i], poly[(i+1)%n]).toLeft(p);
            int d1 = sgn(poly[i].y - p.y);
            int d2 = sgn(poly[(i + 1) % n].y - p.y);
            if (k>0 && d1<=0 && d2>0) { wn++; }
            if (k<0 && d2<=0 && d1>0) { wn--; }
        }
        return wn;
    }
};

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    int n, m;
    cin >> n;
    Polygon poly(n);
    cin >> m;
    while (m--) {
        Point p; p.in();
        int res = poly.inPolygon(p);
        if (res == -inf) { cout << "EDGE\n"; }
        else             { cout << res << "\n"; }
    }
}
```
