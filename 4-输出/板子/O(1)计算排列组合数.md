$C_n^k=\dfrac{n!}{k!(n-k)!}$
```cpp
// 预处理阶乘和阶乘的逆元
vector<ll> fac(n+1), Inv(n+1);
fac[0] = Inv[0] = 1;
for (int i = 1; i <= n; ++i) {
	fac[i] = fac[i-1]*i%MOD;
	Inv[i] = qpow(fac[i]);
}

// 计算
auto C=[&](ll n, ll k) {
    return n==0? 1: fac[n] * Inv[k] % MOD * Inv[n-k] % MOD;
};
auto P=[&](ll n, ll k) {
    return n==0? 1: fac[n] * Inv[n-k] % MOD;
}
```
