void Dijkstra()
{
    priority_queue <pii, vector<pii>, greater<pii>> q;
    for (int i = 1; i <= n; ++i) { dis[i] = inf; }
    dis[1] = 0;
    q.push( {0,1} );
    while (!q.empty()) {
        int u = q.top().second; q.pop();
        if (vis[u]) { continue; }
        vis[u] = true;
        for (auto [v,c]: G[u]) {
            if (dis[v] > dis[u] + 1) {
                dis[v] = dis[u] + 1;
                if (!vis[v]) { q.push( {dis[v], v} ); }
            }
        }
    }
    #ifndef ONLINE_JUDGE
    for (int i = 1; i <= n; ++i) { printf("dis[%d]=%d\n", i, dis[i]); }
    #endif
}
